---
title: "Get Scan"
api: "GET /api/scans/{id}"
description: "Retrieve details of a specific scan by ID"
---

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier (UUID) of the scan
</ParamField>

## Response

Returns a scan object with the following fields:

<ResponseField name="id" type="string">
  Unique identifier (UUID) of the scan
</ResponseField>

<ResponseField name="repo_id" type="string">
  ID of the repository being scanned
</ResponseField>

<ResponseField name="scan_type" type="string">
  Type of scan performed
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the scan. Possible values:
  - `queued` - Scan is waiting to start
  - `ingesting` - Ingesting repository code
  - `ingested` - Ingestion complete
  - `modeling` - Building threat model
  - `modeled` - Threat model complete
  - `static_analysis` - Running static analysis
  - `hunting` - AI-powered vulnerability hunting
  - `hunted` - Hunting complete
  - `validating` - Validating findings
  - `validated` - Validation complete
  - `reporting` - Generating report
  - `completed` - Scan finished successfully
  - `failed` - Scan failed
  - `cancelled` - Scan was cancelled
</ResponseField>

<ResponseField name="commit_sha" type="string" optional>
  Git commit SHA being scanned
</ResponseField>

<ResponseField name="base_commit_sha" type="string" optional>
  Base commit SHA for differential scans
</ResponseField>

<ResponseField name="parent_scan_id" type="string" optional>
  ID of the parent scan (for incremental scans)
</ResponseField>

<ResponseField name="triggered_by" type="string" optional>
  User ID who triggered the scan
</ResponseField>

<ResponseField name="finding_count" type="integer">
  Total number of findings discovered
</ResponseField>

<ResponseField name="critical_count" type="integer">
  Number of critical severity findings
</ResponseField>

<ResponseField name="high_count" type="integer">
  Number of high severity findings
</ResponseField>

<ResponseField name="medium_count" type="integer">
  Number of medium severity findings
</ResponseField>

<ResponseField name="low_count" type="integer">
  Number of low severity findings
</ResponseField>

<ResponseField name="started_at" type="string" optional>
  ISO 8601 timestamp when the scan started
</ResponseField>

<ResponseField name="completed_at" type="string" optional>
  ISO 8601 timestamp when the scan completed
</ResponseField>

<ResponseField name="error_message" type="string" optional>
  Error message if the scan failed
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp when the scan was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp when the scan was last updated
</ResponseField>
